# 取模mod

mod(36,-10)=-4  即：36 除以 10 的整数商为 3，加 1 后为 4；其与除数之积为 40；再与被数之差为（40-36=4）；取除数的符号。所以值为 -4。

# 取余rem

mod(5,2)
ans =1                   % 除数是正，余数就是正
mod(-5,2)
ans =1

mod(5,-2)
ans =-1                  % 除数是负，余数就是负
mod(-5,-2)
ans =-1      

rem(5,2)                % 用 rem 时，不管除数是正是负，余数的符号与被除数的符号相同
ans =1                   % 被除数是正，余数就是正
rem(5,-2); 
ans =1

rem(-5,2)
ans =-1                 % 被除数是负， 余数就是负
rem(-5,-2)
ans =-1
